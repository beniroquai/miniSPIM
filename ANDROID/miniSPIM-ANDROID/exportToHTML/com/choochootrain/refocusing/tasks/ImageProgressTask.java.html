3    import android.app.ProgressDialog;

7    public abstract class ImageProgressTask extends AsyncTask<Float, Integer, Void> { 
8        protected Context context; 
9        protected ProgressDialog progressDialog; 
10    
11       public ImageProgressTask(Context context) { 
12           this.context = context; 
13           this.progressDialog = new ProgressDialog(context); 
14           this.progressDialog.setProgressStyle(ProgressDialog.STYLE_HORIZONTAL); 
15       } 
16    
17       @Override 
18       protected void onProgressUpdate(Integer... progress){ 
19           if (progress.length > 1) { 
20               if (progress[0] >= 0) 
21                   progressDialog.setProgress(progress[0]); 
22               if (progress[1] >= 0) 
23                   progressDialog.setSecondaryProgress(progress[1]); 
24           } 
25       } 
26    
27       @Override 
28       protected void onPreExecute() { 
29           progressDialog.show(); 
30       } 
31    
32       @Override 
33       protected void onPostExecute(Void result) { 
34           if (progressDialog.isShowing()) 
35               progressDialog.dismiss();
