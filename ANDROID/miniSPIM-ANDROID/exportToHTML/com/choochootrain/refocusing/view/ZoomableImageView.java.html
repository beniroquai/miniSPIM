1    package com.choochootrain.refocusing.view; 
2     
3    import android.content.Context; 
4    import android.graphics.Bitmap; 
5    import android.graphics.Canvas; 
6    import android.graphics.drawable.BitmapDrawable; 
7    import android.graphics.drawable.Drawable; 
8    import android.util.AttributeSet; 
9    import android.view.MotionEvent; 
10   import android.view.ScaleGestureDetector; 
11   import android.view.View; 
12    
13   public class ZoomableImageView extends View { 
14       private static final int INVALID_POINTER_ID = -1; 
15       private static final float MIN_SCALE = 0.05f; 
16       private static final float MAX_SCALE = 20.0f; 
17    
18       private Drawable image; 
19       private float xPos; 
20       private float yPos; 
21       private float lastXPos; 
22       private float lastYPos; 
23       private int activePointerId = INVALID_POINTER_ID; 
24    
25       private ScaleGestureDetector scaleDetector; 
26       private float scaleFactor = 1.f; 
27    
28       public ZoomableImageView(Context context) { 
29           this(context, null, 0); 
30       } 
31    
32       public ZoomableImageView(Context context, AttributeSet attrs) { 
33           this(context, attrs, 0); 
34       } 
35    
36       public ZoomableImageView(Context context, AttributeSet attrs, int defStyle) { 
37           super(context, attrs, defStyle); 
38           scaleDetector = new ScaleGestureDetector(context, new ScaleListener()); 
39       } 
40    
41       public void setImage(Bitmap bmp) { 
42           image = new BitmapDrawable(getContext().getResources(), bmp); 
43           image.setBounds(0, 0, image.getIntrinsicWidth(), image.getIntrinsicHeight()); 
44    
45           invalidate(); 
46       } 
47    
48       @Override 
49       public boolean onTouchEvent(MotionEvent ev) { 
50           // pass all events through ScaleGestureDetector 
51           scaleDetector.onTouchEvent(ev); 
52    
53           final int action = ev.getAction(); 
54    
55           switch (action & MotionEvent.ACTION_MASK) { 
56               case MotionEvent.ACTION_DOWN: 
57                   lastXPos = ev.getX(); 
58                   lastYPos = ev.getY(); 
59                   activePointerId = ev.getPointerId(0); 
60                   break; 
61               case MotionEvent.ACTION_MOVE: 
62                   final int pointerIndex = ev.findPointerIndex(activePointerId); 
63                   float x = ev.getX(pointerIndex); 
64                   float y = ev.getY(pointerIndex); 
65    
66                   if (!scaleDetector.isInProgress()) { 
67                       xPos += x - lastXPos; 
68                       yPos += y - lastYPos; 
69    
70                       invalidate(); 
71                   } 
72    
73                   lastXPos = x; 
74                   lastYPos = y; 
75    
76                   break; 
77               case MotionEvent.ACTION_POINTER_UP: 
78                   final int idx = (ev.getAction() & MotionEvent.ACTION_POINTER_INDEX_MASK) 
79                           >> MotionEvent.ACTION_POINTER_INDEX_SHIFT; 
80                   int pointerId = ev.getPointerId(idx); 
81                   if (pointerId == activePointerId) { 
82                       // This was our active pointer going up. Choose a new 
83                       // active pointer and adjust accordingly. 
84                       final int newPointerIndex = idx == 0 ? 1 : 0; 
85                       lastXPos = ev.getX(newPointerIndex); 
86                       lastYPos = ev.getY(newPointerIndex); 
87                       activePointerId = ev.getPointerId(newPointerIndex); 
88                   } 
89                   break; 
90               case MotionEvent.ACTION_UP: 
91                   activePointerId = INVALID_POINTER_ID; 
92                   break; 
93               case MotionEvent.ACTION_CANCEL: 
94                   activePointerId = INVALID_POINTER_ID; 
95                   break; 
96           } 
97    
98           return true; 
99       } 
100   
101      @Override 
102      public void onDraw(Canvas canvas) { 
103          super.onDraw(canvas); 
104   
105          if (image != null) { 
106              canvas.save(); 
107              canvas.translate(xPos, yPos); 
108              canvas.scale(scaleFactor, scaleFactor); 
109              canvas.translate(-image.getIntrinsicWidth()/2, -image.getIntrinsicHeight()/2); 
110              image.draw(canvas); 
111              canvas.restore(); 
112          } 
113      } 
114   
115      private class ScaleListener extends ScaleGestureDetector.SimpleOnScaleGestureListener { 
116          @Override 
117          public boolean onScale(ScaleGestureDetector detector) { 
118              scaleFactor *= detector.getScaleFactor(); 
119              scaleFactor = Math.max(MIN_SCALE, Math.min(scaleFactor, MAX_SCALE)); 
120   
121              invalidate(); 
122              return true; 
123          } 
124      } 
125  }
